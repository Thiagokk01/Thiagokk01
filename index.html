function updateCharts(country) { if (!chartCo2 || !chartTrees || !chartBiodiversity) return; // Atualiza as cores dos gráficos para destacar o país atual const highlightColor = "#f39c12"; const defaultColor = "#27ae60"; const index = countriesData.findIndex(c => c.name === country.name); chartCo2.data.datasets[0].backgroundColor = countriesData.map((_, i) => i === index ? highlightColor : defaultColor ); chartCo2.update(); chartTrees.data.datasets[0].borderColor = countriesData.map((_, i) => i === index ? highlightColor : defaultColor ); chartTrees.update(); chartBiodiversity.data.datasets[0].borderColor = highlightColor; chartBiodiversity.update(); }
